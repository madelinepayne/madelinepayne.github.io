    courses TA'd by this person. Also contains various functions for the
    manipulation of each list.

#include "Person.h"

// purpose: constructor
// args: name
// rets: none
Person::Person(string name)
{
    this->name = name;
}

// purpose: destructor
// args: none
// rets: none
Person::~Person()
{

}

// purpose: name accessor
// args: none
// rets: string
string& Person::get_name()
{
    return name;
}

// purpose: finds the path between two students (see readme for algorithm)
// args: Person, Path, vector<Path>
// rets: none
void Person::find_path(Person *b, Path path, vector<Path> &found_paths)
{
    PathElement new_element;
    new_element.person = this;
    new_element.course = NULL;
    if (path.get_size() > 0 && this == b) {
        // add our path to the list of found paths!!!
        path.add(new_element);
        found_paths.push_back(path);
    } else if (!path.in_list(new_element)) {
        // go and check the roster in each course that this person TA'd to see
        // if you can find the Person b in there
        path.add(new_element);
        for (int i = 0; i < taed.get_count(); i++) {
            taed.courses[i]->find_path(b, path, found_paths);
        }
    }
}
