    By: Madeline Payne
    This class contains a pathelement struct which makes up one member of any
    path that is being made between two people. It contains other functions
    related to manipulating the path elements.

 */
#include <iostream>
#include <string>
#include "Path.h"
#include "Course.h"
#include "Person.h"
using namespace std;

// purpose: adds given element to the end of the path
// args: PathElement to be added
// rets: none
void Path::add(PathElement new_element)
{
    elements.push_back(new_element);
}

// purpose: accessor for size of path
// args: none
// rets: int (size)
int Path::get_size()
{
    return elements.size();
}

// purpose: checks whether given element is in the list
// args: element to checks
// rets: true or false
bool Path::in_list(PathElement e)
{

    for (unsigned long i = 0; i < elements.size(); i++) {
        PathElement *pe = &elements[i];
        if (pe->person == e.person && pe->course == e.course)
        {
            return true;
        }
    }
    return false;
}

// purpose: prints out the path
// args: none
// rets: none
void Path::print_path()
{
    const unsigned long size = elements.size();
    for (unsigned long i = 0; i < size - 1; i++) {
        if (elements[i].person != NULL) {
            cout << elements[i].person->get_name() << " +- ";
        } else {
            cout << elements[i].course->get_ID() << " -> ";
        }
    }

    cout << elements[size - 1].person->get_name();
    cout << endl;
}
